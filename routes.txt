ROUTES

GET /
Input: none
Response: { message: "Express server is running." }

GET /health
Input: none
Response: { status: "ok" }

POST /pdf/wordcount
Input: multipart/form-data with field "file" containing a PDF document
Response: { wordCount: number, uniqueWordCount: number, frequencies: { [word]: count }, topWords: [{ word: string, count: number }], mapResults: [{ worker, ok, status, response }] }
Error Responses: 400 for missing/invalid file, 500 on processing errors

POST /map/frequencies (backend1:7002 and backend2:7003)
Input: JSON { source: string, createdAt: string, words: string[] }
Response: { worker: string, batchId: number, wordCount: number, uniqueWordCount: number, frequencies: { [word]: count } }

POST /store/frequencies (backend1:7002 and backend2:7003)
Input: JSON { source: string, createdAt: string, frequencies: { [word]: count } }
Response: { message: string, batchId: number, uniqueWords: number, totalBatches: number }

GET /store/frequencies (backend1:7002 and backend2:7003)
Input: none
